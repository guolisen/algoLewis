# 最长公共子串(双序列dp)

### 77 · Longest Common Subsequence
f(i, j) 是1-i和i- j两个子序列的LCS
```
class Solution {
public:
    int longestCommonSubsequence(string &A, string &B) {
        int m = A.size();
        int n = B.size();
        A = " " + A;
        B = " " + B;
        vector<vector<int>> f(m + 1, vector<int>(n + 1, 0));
        
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                f[i][j] = max(f[i - 1][j], f[i][j - 1]);
                if (A[i] == B[j])
                {
                    f[i][j] = max(f[i][j], f[i - 1][j - 1] + 1);
                }
            }
        }
        return f[m][n];
    }
};
```

### 119 · Edit Distance

```
class Solution {
public:
    int minDistance(string &word1, string &word2) {
        int m = word1.length();
        int n = word2.length();
        word1 = " " + word1;
        word2 = " " + word2;
        vector<vector<int>> f(m + 1, vector<int>(n + 1, 0));
        for (int i = 1; i <= m; i++)
        {
            f[i][0] = i;
        }
        for (int i = 1; i <= n; i++)
        {
            f[0][i] = i;
        }
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                f[i][j] = min(f[i - 1][j], f[i][j - 1]) + 1;
                if (word1[i] == word2[j])
                {
                    f[i][j] = min (f[i][j], f[i - 1][j - 1]);
                }
                else
                {
                    f[i][j] = min (f[i][j], f[i - 1][j - 1] + 1);
                }
            }
        }
        return f[m][n];
    }
};
```
### 1076 · Minimum ASCII Delete Sum for Two Strings
```
class Solution {
public:
    int minimumDeleteSum(string &s1, string &s2) {
        int m = s1.length();
        int n = s2.length();
        s1 = " " + s1;
        s2 = " " + s2;
        vector<vector<int>> f(m + 1, vector<int>(n + 1, 0));
        f[0][0] = 0;
        for (int i = 1; i <= m; i++)
        {
            f[i][0] = f[i - 1][0] + s1[i];
        }
        for (int i = 1; i <= n; i++)
        {
            f[0][i] = f[0][i - 1] + s2[i];
        }
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                f[i][j] = min(f[i - 1][j] + s1[i], f[i][j - 1] + s2[j]);
                if (s1[i] == s2[j])
                {
                    f[i][j] = min(f[i][j], f[i - 1][j - 1]);
                }
                else
                {
                    f[i][j] = min(f[i][j], f[i - 1][j - 1] + s1[i] + s2[j]);
                }
            }
        }
        return f[m][n];
    }
};
```

### 515 · Paint House
```
class Solution {
public:
    int minCost(vector<vector<int>> &costs) {
        if (costs.empty())
            return 0;
        int m = costs.size();
        vector<vector<int>> f(m, vector<int>(3, 0));
        f[0][0] = costs[0][0];
        f[0][1] = costs[0][1];
        f[0][2] = costs[0][2];
        for (int i = 1; i < m; i++)
        {
            for (int j = 0; j < 3; j++)
            {
                f[i][j] = min(f[i - 1][(j + 1) % 3], f[i - 1][(j + 2) % 3]) + costs[i][j];
            }
        }
        int res = INT_MAX;
        for (int i = 0; i < 3; i++)
        {
            res = min(res, f[m - 1][i]);
        }

        return res;
    }
};
```
