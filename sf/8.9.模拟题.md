```
class ListNode:
    """链表节点，存储哈希集合的元素"""
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class MyHashSet:
    def __init__(self):
        """初始化哈希集合：桶数量选1009（质数，减少冲突）"""
        self.bucket_size = 1009
        self.buckets = [ListNode() for _ in range(self.bucket_size)]  # 虚拟头节点
    
    def _hash(self, key: int) -> int:
        """哈希函数：计算key对应的桶索引"""
        return key % self.bucket_size

    def add(self, key: int) -> None:
        """添加元素到集合（已存在则不操作）"""
        idx = self._hash(key)
        cur = self.buckets[idx]  # 从虚拟头节点开始遍历
        # 遍历链表，检查是否已存在
        while cur.next:
            if cur.next.val == key:
                return  # 已存在，直接返回
            cur = cur.next
        # 不存在则新增节点
        cur.next = ListNode(key)

    def remove(self, key: int) -> None:
        """从集合中删除元素（不存在则不操作）"""
        idx = self._hash(key)
        cur = self.buckets[idx]
        # 遍历链表，找到目标节点的前驱
        while cur.next:
            if cur.next.val == key:
                cur.next = cur.next.next  # 删除节点
                return
            cur = cur.next

    def contains(self, key: int) -> bool:
        """判断元素是否在集合中"""
        idx = self._hash(key)
        cur = self.buckets[idx].next  # 跳过虚拟头节点
        while cur:
            if cur.val == key:
                return True
            cur = cur.next
        return False
```

```
        idx = self._hash(key)
        # 线性探测找空位置
        while self.table[idx] is not None:
            if self.table[idx] == key:
                return  # 已存在，无需添加
            idx = (idx + 1) % self.size  # 向后探测
        self.table[idx] = key

    def remove(self, key: int) -> None:
        idx = self._hash(key)
        # 线性探测找目标key

            if self.table[idx] == key:
                self.table[idx] = -1  # 标记为已删除
                return
            idx = (idx + 1) % self.size

            if self.table[idx] == key:
```

```
class DListNode:
    def __init__(self, key, val):
        self.key = key 
        self.val = val 
        self.size = 0
        self.next = None 
        self.prev = None 
class DList:
    def __init__(self):
        self.head = DListNode(0, 0)
        self.tail = DListNode(-1, -1)
        self.head.next = self.tail
        self.tail.prev = self.head 

    def addToHead(self, node):
        node.next = self.head.next 
        self.head.next.prev = node
        self.head.next = node 
        node.prev = self.head 

    def removeNode(self, node):
        node.prev.next = node.next 
        node.next.prev = node.prev

    def removeTail(self):
        tailNode = self.tail.prev
        self.removeNode(tailNode)
        return tailNode

class LRUCache:

    def __init__(self, capacity: int):
        self.data = DList()
        self.capacity = capacity
        self.size = 0
        self.cache = dict()

    def get(self, key: int) -> int:
        if key not in self.cache.keys():
            return -1
        cur = self.cache[key]
        self.data.removeNode(cur)
        self.data.addToHead(cur)
        return cur.val

    def put(self, key: int, value: int) -> None:
        if key in self.cache.keys():
            cur = self.cache[key]
            self.data.removeNode(cur)
            cur.val = value
            self.data.addToHead(cur)
            return   
        newNode = DListNode(key, value)
        self.data.addToHead(newNode)
        self.cache[key] = newNode
        self.size += 1
        if self.size > self.capacity:
            tailNode = self.data.removeTail()
            del self.cache[tailNode.key]
            self.size -= 1
```

```
    @param root: An object of TreeNode, denote the root of the binary tree.
    This method will be invoked first, you should design your own algorithm 
    to serialize a binary tree which denote by a root node to a string which
    can be easily deserialized by your own "deserialize" method later.
    """
    def serialize(self, root):
        if root == None:
            return None
        q = collections.deque([root])
        res = []
        while len(q) > 0:
            cur = q.popleft()
            if cur.val != -1:
                res.append(str(cur.val))
            else:
                res.append("#")
                continue
            l = cur.left if cur.left != None else TreeNode(-1)
            q.append(l)
            r = cur.right if cur.right != None else TreeNode(-1)
            q.append(r)
        return " ".join(res)

    """
    @param data: A string serialized by your serialize method.
    This method will be invoked second, the argument data is what exactly
    you serialized at method "serialize", that means the data is not given by
    system, it's given by your own serialize method. So the format of data is
    designed by yourself, and deserialize it here as you serialize it in 
    "serialize" method.
    """
    def deserialize(self, data):
        if data == None:
            return None
        lnodes = [TreeNode(int(n)) if n != "#" else None for n in data.split()]
        root = lnodes[0]
        nodes = [root]
        slow = 0
        fast = 1
        while slow < len(nodes):
            cur = nodes[slow]
            cur.left = lnodes[fast]
            cur.right = lnodes[fast + 1]

            if cur.left:
                nodes.append(cur.left)
            if cur.right:
                nodes.append(cur.right)
            slow += 1
            fast += 2
```

```
class Solution {

    unordered_map<char, int> pr{{'+', 1}, {'-', 1}, {'*', 2}, {'/', 2}};
    stack<long long> nums;
    stack<char> op;

    void eval(){
        if(nums.size() < 2 || op.empty())
            return;
        long long b = nums.top(); nums.pop();
        long long a = nums.top(); nums.pop();
        long long c = op.top(); op.pop();
        long long res;
        if (c == '+') res = a + b;
        else if (c == '-') res = a - b;
        else if (c == '*') res = a * b;
        else res = a / b;
        nums.push(res);
    }

    void replace(string &s, string a, string b) {
        int pos = s.find(a), m = a.size();
        while (pos != -1) {
            s.replace(pos, m, b);
            pos = s.find(a);
        }
    }

    int calculate(string s) {
        // 处理 s 使之合法
        // 去空格
        replace(s, " ", "");
        // (- 替换为 (0-
        replace(s, "(-", "(0-");
        // (+ 替换为 (0+
        replace(s, "(+", "(0+");
        // 在 nums 中先放个 0 预防开始的符号
        nums.push(0);

        for (int i = 0; i < s.size(); ++ i) {
            char c = s[i];
            if (isdigit(c)) {
                int j = i; long long num = 0;
                while (j < s.size() && isdigit(s[j]))
                    num = num * 10 + (s[j++] - '0');
                i = j - 1;
                nums.push(num);
            } else if (c == '(') {
                op.push('(');
            }
            else if (c == ')') {
                while (op.top() != '(') eval();
                op.pop();
            } else {
                while (op.size() && op.top() != '(' && pr[op.top()] >= pr[c]) eval();
                op.push(c);
            }
        }
        while (op.size()) eval();
        return nums.top();
    }
};
```

```
    def __init__(self):
        self.head = ListNode(-1)  # 虚拟头
        self.tail = ListNode(-1)  # 虚拟尾
        self.head.next = self.tail
        self.tail.prev = self.head  # 初始化：head ↔ tail
        self.size = 0

    def get(self, index: int) -> int:
        if index < 0 or index >= self.size:
            return -1
        # 选择更近的遍历方向找目标节点
        if index + 1 < self.size - index:
            cur = self.head 
            for _ in range(index + 1):
                cur = cur.next 
        else:
            cur = self.tail
            for _ in range(self.size - index):
                cur = cur.prev 
        return cur.val  

    def addAtHead(self, val: int) -> None:
        self.addAtIndex(0, val)

    def addAtTail(self, val: int) -> None:
        self.addAtIndex(self.size, val)

    def addAtIndex(self, index: int, val: int) -> None:
        if index < 0 or index > self.size:
            return
        newNode = ListNode(val)
        if index < self.size - index:
            # 正向找插入位置的前驱
            pred = self.head 
            for _ in range(index):
                pred = pred.next 
            succ = pred.next 
            # 插入新节点
            newNode.prev = pred
            newNode.next = succ
            pred.next = newNode
            succ.prev = newNode
        else:
            # 反向找插入位置的后继
            succ = self.tail  
            for _ in range(self.size - index):
                succ = succ.prev 
            pred = succ.prev 
            # 插入新节点
            newNode.prev = pred
            newNode.next = succ
            pred.next = newNode
            succ.prev = newNode
        self.size += 1

    def deleteAtIndex(self, index: int) -> None:
        if index < 0 or index >= self.size:
            return 
        
        # 核心：先精准找到要删除的节点 cur
        if index + 1 < self.size - index:
            # 正向遍历找删除节点（和get方法逻辑完全一致）
            cur = self.head 
            for _ in range(index + 1):  # index+1步 → 直接到删除节点
                cur = cur.next 
        else:
            # 反向遍历找删除节点（和get方法逻辑完全一致）
            cur = self.tail
            for _ in range(self.size - index):  # size-index步 → 直接到删除节点
                cur = cur.prev 
        
        # 你想要的删除逻辑：让cur的前后节点互相指向，跳过cur
        cur.next.prev = cur.prev  # 后继节点的前驱 → cur的前驱
        cur.prev.next = cur.next  # 前驱节点的后继 → cur的后继
        self.size -= 1  # 别忘了更新长度
```





# 138. Copy List with Random Pointer



```
"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if head == None:
            return None
        h = collections.defaultdict(Node)
        arr = []
        cur = head 
        while cur:
            h[cur] = Node(cur.val)
            arr.append(cur)
            cur = cur.next 
        m = len(arr)
        for i in range(m - 1):
            h[arr[i]].next = h[arr[i + 1]]
            if arr[i].random in arr:
                h[arr[i]].random = h[arr[i].random]
        h[arr[m - 1]].next = None
        if arr[m - 1].random in arr:
            h[arr[m - 1]].random = h[arr[m - 1].random]
        return h[head]
```

# 460. LFU 缓存

```
class LFUCache {
public:
    struct Node {
        Node *left, *right;
        int key, val;
        Node(int _key, int _val) {
            key = _key, val = _val;
            left = right = NULL;
        }
    };
    struct Block {
        Block *left, *right;
        Node *head, *tail;
        int cnt;
        Block(int _cnt) {
            cnt = _cnt;
            left = right = NULL;
            head = new Node(-1, -1), tail = new Node(-1, -1);
            head->right = tail, tail->left = head;
        }
        ~Block() {
            delete head;
            delete tail;
        }
        void insert(Node* p) {
            p->right = head->right;
            head->right->left = p;
            p->left = head;
            head->right = p;
        }
        void remove(Node* p) {
            p->left->right = p->right;
            p->right->left = p->left;
        }
        bool empty() {
            return head->right == tail;
        }
    }*head, *tail;
    int n;
    unordered_map<int, Block*> hash_block;
    unordered_map<int, Node*> hash_node;

    void insert(Block* p) {  // 在p的右侧插入新块，cnt是p->cnt + 1
        auto cur = new Block(p->cnt + 1);
        cur->right = p->right;
        p->right->left = cur;
        p->right = cur;
        cur->left = p;
    }

    void remove(Block* p) {
        p->left->right = p->right;
        p->right->left = p->left;
        delete p;
    }

    LFUCache(int capacity) {
        n = capacity;
        head = new Block(0), tail = new Block(INT_MAX);
        head->right = tail, tail->left = head;
    }

    int get(int key) {
        if (hash_block.count(key) == 0) return -1;
        auto block = hash_block[key];
        auto node = hash_node[key];
        block->remove(node);
        if (block->right->cnt != block->cnt + 1) insert(block);
        block->right->insert(node);
        hash_block[key] = block->right;
        if (block->empty()) remove(block);
        return node->val;
    }

    void put(int key, int value) {
        if (!n) return;
        if (hash_block.count(key)) {
            hash_node[key]->val = value;
            get(key);
        } else {
            if (hash_block.size() == n) {
                auto p = head->right->tail->left;
                head->right->remove(p);
                if (head->right->empty()) remove(head->right);
                hash_block.erase(p->key);
                hash_node.erase(p->key);
                delete p;
            }
            auto p = new Node(key, value);
            if (head->right->cnt > 1) insert(head);
            head->right->insert(p);
            hash_block[key] = head->right;
            hash_node[key] = p;
        }
    }
};
```
